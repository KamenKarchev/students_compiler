program {
    main() -> void {
        int[] arr = int[10];
        int i = 0;
        while (i < 10) {
            arr[i] = i * 3 + 1; // sample data
            i = i + 1;
        }

        int max = arr[0];
        i = 1;
        while (i < 10) {
            if (arr[i] > max) {
                max = arr[i];
            }
            i = i + 1;
        }
        print("Max =", max);
    }
}
